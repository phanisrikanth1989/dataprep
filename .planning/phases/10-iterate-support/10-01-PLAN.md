    ```python
    def execute(self, input_data: Optional[pd.DataFrame] = None) -> Dict[str, Any]:
        """Iterate-specific execute(). Skips data-pipeline lifecycle steps
        (output schema validation, REJECT routing, batch/streaming dispatch).
        Keeps: status transitions, _original_config snapshot/restore (EXEC-06),
        _validate_config, _resolve_expressions, _update_global_map.
        After this returns, the Executor's `_execute_iterate_body` runs the loop.
        """
        import copy
        import time
        self.status = ComponentStatus.RUNNING
        start = time.time()
        try:
            # Per-iteration config freshness (EXEC-06; D-I2)
            self.config = copy.deepcopy(self._original_config)
            self._validate_config()
            self._resolve_expressions()
            self.die_on_error = bool(self.config.get("die_on_error", False))

            # Hook: prepare (D-A5.1)
            self.prepare()

            # Hook: prepare_iterations (D-A5.2) -- subclass populates iteration_iter
            self.iteration_iter = self.prepare_iterations(input_data)

            # NOTE: Executor drives the loop via _execute_iterate_body. We do NOT
            # iterate here. Stats finalization happens after the loop via
            # _update_global_map() called by Executor at end of iterate body.

            # Initial status; Executor flips to SUCCESS after iterate completes
            self.execution_time = time.time() - start
            return {
                "main": None,      # iterate components produce no main DataFrame
                "reject": None,    # reject is accumulated by Executor across iterations
                "stats": self.stats,
            }
        except ConfigurationError:
            self.status = ComponentStatus.FAILED
            raise
        except Exception as exc:
            self.status = ComponentStatus.FAILED
            self.execution_time = time.time() - start
            raise ComponentExecutionError(self.id, str(exc), cause=exc)
    ```

    ```python
        """Configurable test stub for the executor iterate loop and base-class tests.

        Config keys:
            items (list[Any]): items to yield from prepare_iterations
            globalmap_key_prefix (str): prefix for keys written by set_iteration_globalmap
                (default "TEST_")
            stop_after (int | None): if set, should_stop returns True at this 1-based index
            fail_at (int | None): if set, raises ComponentExecutionError at this 1-based index
        """

        def _validate_config(self) -> None:
            items = self.config.get("items", [])
            if not isinstance(items, list):
                from src.v1.engine.exceptions import ConfigurationError
                raise ConfigurationError(f"[{self.id}] 'items' must be a list")

        def prepare_iterations(self, input_data=None):
            items = self.config.get("items", [])
            self.total_iterations = len(items)
            return iter(items)

        def set_iteration_globalmap(self, item) -> None:
            if self.global_map is None:
                return
            prefix = self.config.get("globalmap_key_prefix", "TEST_")
            # The IterateStubComponent emits one key per iteration with the index it was
            # called for; tests inspect global_map after each call.
            idx = getattr(item, "index", None)
            value = getattr(item, "value", item)
            if idx is None:

                self._stub_counter = getattr(self, "_stub_counter", 0) + 1
                idx = self._stub_counter
            self.global_map.put(f"{prefix}{idx}", value)

        def should_stop(self, item, index: int) -> bool:
            stop_after = self.config.get("stop_after")
            return stop_after is not None and index > stop_after
    ```
