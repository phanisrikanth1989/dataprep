    """Integration tests for tFileExist (Phase 10-08, verify-only).

    tFileExist is already GREEN per Phase 9 audit. This test verifies
    ITER-08 (file_name vs file_path keys) and ITER-09 ({id}_EXISTS, {id}_FILENAME
    globalMap vars) in a real RUN_IF branching scenario (D-K3).

    No engine code changes in Phase 10 for tFileExist.
    """
    import json
    from pathlib import Path

    import pytest

    from src.v1.engine.engine import ETLEngine, run_job


    def _make_file_exist_run_if_config(
        file_path_value: str,
        config_key: str = "file_name",
        downstream_marker_path: str = "/tmp/marker.out",
    ) -> dict:
        """Build a job config with:
            tFileExist_1 (config[config_key] = file_path_value)
            -> RUN_IF (condition checks tFileExist_1_EXISTS) -> downstream subjob

        Downstream subjob writes a marker file when triggered (so we can detect
        whether the RUN_IF branch fired without depending on Java).
        """
        return {
            "components": [
                {
                    "id": "tFileExist_1",
                    "type": "tFileExist",
                    "config": {config_key: file_path_value},
                    "inputs": [],
                    "outputs": [],
                    "subjob_id": "subjob_1",
                },
                # A simple downstream component that runs when RUN_IF fires.
                # Use a Python-routine-friendly component or a tFileOutputDelimited
                # writing to a marker path. CHOOSE the simplest verified component
                # available in the engine. Suggested: tJava with a simple python_code
                # writing the marker file (Phase 8 code components support this).
                # If that's heavy, use a StubComponent registered in conftest -- but
                # StubComponent needs to be exposed to engine COMPONENT_REGISTRY for
                # this to work, which is test-time only.
                # Concrete choice: tJava with python_code = "open(MARKER, 'w').close()"
                {
                    "id": "tJava_marker",
                    "type": "tJava",  # or whatever pure-Python code component exists
                    "config": {
                        "python_code": (
                            "import pathlib; "
                            "pathlib.Path('{marker}').write_text('fired')"
                        ).format(marker=downstream_marker_path),
                        "die_on_error": False,
                    },
                    "inputs": [],
                    "outputs": [],
                    "subjob_id": "subjob_2",
                },
            ],
            "flows": [],
            "triggers": [
                {
                    "from": "tFileExist_1",
                    "to": "tJava_marker",
                    "trigger_type": "RunIf",
                    "condition": "globalMap.get('tFileExist_1_EXISTS') == True",
                    # Note: condition format depends on TriggerManager's expected
                    # syntax. Read trigger_manager.py to confirm whether Python or
                    # Java syntax is expected here.
                },
            ],
            "subjobs": {
                "subjob_1": {"components": ["tFileExist_1"]},
                "subjob_2": {"components": ["tJava_marker"]},
            },
            "context": {},
        }


    class TestFileExistConfigKeyAliases:
        """ITER-08: file_name vs file_path keys both accepted."""

        def test_file_name_key_accepted(self, tmp_path):
            target = tmp_path / "exists.txt"
            target.write_text("x")
            config = _make_file_exist_run_if_config(
                str(target), config_key="file_name",
                downstream_marker_path=str(tmp_path / "marker.out"),
            )
            stats = ETLEngine().run(config)  # adapt to actual run signature
            assert stats["status"] in ("success", "completed")
            assert stats["globalmap"].get("tFileExist_1_EXISTS") is True

        def test_file_path_key_accepted_legacy(self, tmp_path):
            target = tmp_path / "exists.txt"
            target.write_text("x")
            config = _make_file_exist_run_if_config(
                str(target), config_key="file_path",
                downstream_marker_path=str(tmp_path / "marker.out"),
            )
            stats = ETLEngine().run(config)
            assert stats["status"] in ("success", "completed")
            assert stats["globalmap"].get("tFileExist_1_EXISTS") is True


    class TestGlobalMapVariables:
        """ITER-09: {id}_EXISTS and {id}_FILENAME both set."""

        def test_exists_true_for_present_file(self, tmp_path):
            target = tmp_path / "x.txt"
            target.write_text("x")
            config = _make_file_exist_run_if_config(
                str(target), downstream_marker_path=str(tmp_path / "marker.out"),
            )
            stats = ETLEngine().run(config)
            gm = stats["globalmap"]
            assert gm.get("tFileExist_1_EXISTS") is True
            assert gm.get("tFileExist_1_FILENAME") == str(target)

        def test_exists_false_for_missing_file(self, tmp_path):
            missing = tmp_path / "missing.txt"
            config = _make_file_exist_run_if_config(
                str(missing), downstream_marker_path=str(tmp_path / "marker.out"),
            )
            stats = ETLEngine().run(config)
            gm = stats["globalmap"]
            assert gm.get("tFileExist_1_EXISTS") is False
            assert gm.get("tFileExist_1_FILENAME") == str(missing)


    class TestRunIfBranching:
        """D-K3: tFileExist + RUN_IF -> downstream subjob branching."""

        def test_run_if_branches_when_file_exists(self, tmp_path):
            target = tmp_path / "x.txt"
            target.write_text("x")
            marker = tmp_path / "marker.out"
            config = _make_file_exist_run_if_config(
                str(target), downstream_marker_path=str(marker),
            )
            ETLEngine().run(config)
            assert marker.exists(), "RUN_IF should have fired downstream when file exists"

        def test_run_if_skips_when_file_missing(self, tmp_path):
            missing = tmp_path / "missing.txt"
            marker = tmp_path / "marker.out"
            config = _make_file_exist_run_if_config(
                str(missing), downstream_marker_path=str(marker),
            )
            ETLEngine().run(config)
            assert not marker.exists(), "RUN_IF should NOT fire when file missing"
