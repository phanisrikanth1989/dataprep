    ```python
    """tFileOutputXML engine component (simple/flat XML output).

    Writes a DataFrame to an XML file with one ROW_TAG element per row. Sub-elements
    or attributes per column according to MAPPING.AS_ATTRIBUTE. Optional ROOT_TAGS
    wrapper. Streaming-write via etree.xmlfile (NEVER buffer the full tree per
    Pitfall P-2). Sink contract (S-5): passthrough main; globalMap puts file name +
    line count.

    Per D-D5: hand-authored Job_tFileOutputXML_0.1.item fixture exercises the
    end-to-end pipeline.
    """
    import logging
    import os
    from pathlib import Path
    from typing import Any, Dict, List, Optional

    import pandas as pd
    from lxml import etree

    from ..base_component import BaseComponent
    from ..component_registry import REGISTRY
    from ..exceptions import ConfigurationError, FileOperationError

    logger = logging.getLogger(__name__)


    @REGISTRY.register("FileOutputXML", "tFileOutputXML")
    class FileOutputXML(BaseComponent):
        """Write DataFrame -> XML file (simple/flat).

        Streaming-aware: first chunk opens etree.xmlfile context; subsequent chunks
        reuse the open context; reset() closes contexts on iterate re-execution.
        """

        def __init__(self, *args, **kwargs):
            super().__init__(*args, **kwargs)
            # S-6 streaming-write state (analog of FileOutputDelimited bb5b97f)
            self._streaming_xmlfile_ctx = None        # active etree.xmlfile() ctx mgr
            self._streaming_xmlfile_root_ctx = None   # active xf.element(root_tag) ctx
            self._streaming_write_started: bool = False
            self._streaming_xf = None                  # the entered xf object
            self._streaming_filehandle = None
            self._streaming_total_written: int = 0
            self._streaming_filepath: Optional[str] = None

        def reset(self) -> None:
            """Reset streaming state for iterate re-execution. Closes any leftover ctx."""
            super().reset()
            if self._streaming_xmlfile_root_ctx is not None:
                try:
                    self._streaming_xmlfile_root_ctx.__exit__(None, None, None)
                except Exception:
                    pass
            if self._streaming_xmlfile_ctx is not None:
                try:
                    self._streaming_xmlfile_ctx.__exit__(None, None, None)
                except Exception:
                    pass
            if self._streaming_filehandle is not None:
                try:
                    self._streaming_filehandle.close()
                except Exception:
                    pass
            self._streaming_xmlfile_ctx = None
            self._streaming_xmlfile_root_ctx = None
            self._streaming_write_started = False
            self._streaming_xf = None
            self._streaming_filehandle = None
            self._streaming_total_written = 0
            self._streaming_filepath = None

        @staticmethod
        def _bool(v: Any) -> bool:
            if isinstance(v, str):
                return v.strip().lower() in ("true", "1", "yes")
            return bool(v)

        def _validate_config(self) -> None:
            """Rule 12: presence + type checks only."""
            if not self.config.get("filename"):
                raise ConfigurationError(
                    f"[{self.id}] Missing required config key 'filename'"
                )
            for bool_key in ("input_is_document", "use_dynamic_grouping", "flushonrow",
                              "split", "create", "trim", "advanced_separator",
                              "delete_empty_file", "tstatcatcher_stats"):
                if bool_key in self.config:
                    val = self.config[bool_key]
                    if not isinstance(val, (bool, str, int)):
                        raise ConfigurationError(
                            f"[{self.id}] Config '{bool_key}' must be coercible to bool"
                        )

        def _process(self, input_data: Optional[pd.DataFrame] = None) -> Dict[str, Any]:
            # 1. Read config (with bool coercion)
            filename = self.config["filename"]
            input_is_document = self._bool(self.config.get("input_is_document", False))
            document_col = self.config.get("document_col", "")
            row_tag = self.config.get("row_tag", "row") or "row"
            root_tags = self.config.get("root_tags", []) or []
            mapping = self.config.get("mapping", []) or []
            encoding = self.config.get("encoding", "ISO-8859-15")
            create = self._bool(self.config.get("create", True))
            flushonrow = self._bool(self.config.get("flushonrow", False))
            trim = self._bool(self.config.get("trim", False))
            delete_empty_file = self._bool(self.config.get("delete_empty_file", False))
            split = self._bool(self.config.get("split", False))
            split_every = int(self.config.get("split_every", "1000") or "1000")

            # 2. Resolve filepath
            filepath = Path(filename)

            # 3. CREATE check
            if not create and filepath.exists() and not self._streaming_write_started:
                raise FileOperationError(
                    f"[{self.id}] File exists and create=False: {filename!r}"
                )

            # 4. Empty-input handling
            if input_data is None or len(input_data) == 0:
                if delete_empty_file and filepath.exists():
                    filepath.unlink()
                self._update_stats(0, 0, 0)
                if self.global_map:
                    self.global_map.put(f"{self.id}_FILE_NAME", str(filepath))
                    self.global_map.put(f"{self.id}_NB_LINE", 0)
                return {"main": input_data, "reject": None}

            # 5. Compute attribute-vs-sub-element split from MAPPING
            attr_cols = {m["column"] for m in mapping if self._bool(m.get("as_attribute", False))}

            # 6. Open xmlfile context if first chunk
            if not self._streaming_write_started:
                # Ensure parent dir exists
                if filepath.parent != Path("."):
                    filepath.parent.mkdir(parents=True, exist_ok=True)
                self._streaming_filehandle = open(filepath, "wb")
                self._streaming_xmlfile_ctx = etree.xmlfile(
                    self._streaming_filehandle, encoding=encoding
                )
                self._streaming_xf = self._streaming_xmlfile_ctx.__enter__()
                self._streaming_xf.write_declaration()
                # Apply ROOT_TAGS as nested wrappers (each entry is the next deepest level)
                if root_tags:
                    # Open the outermost root tag and hold its context
                    outermost = root_tags[0].get("name") if isinstance(root_tags[0], dict) else str(root_tags[0])
                    self._streaming_xmlfile_root_ctx = self._streaming_xf.element(outermost)
                    self._streaming_xmlfile_root_ctx.__enter__()
                    # If multiple root_tags, nest the rest (rare; honor first level for v1)
                self._streaming_write_started = True
                self._streaming_filepath = str(filepath)

            # 7. Write rows
            xf = self._streaming_xf
            written = 0
            for _, row in input_data.iterrows():
                if input_is_document:
                    # Whole-row XML doc passthrough
                    doc = row.get(document_col, "")
                    if trim and isinstance(doc, str):
                        doc = doc.strip()
                    if doc:
                        # write the document content as raw XML
                        try:
                            sub_root = etree.fromstring(
                                doc.encode("utf-8") if isinstance(doc, str) else doc
                            )
                            xf.write(sub_root)
                        except etree.XMLSyntaxError as exc:
                            logger.warning("[%s] Skipping malformed XML doc row: %s", self.id, exc)
                            continue
                else:
                    # Build attrs dict from AS_ATTRIBUTE columns
                    attrs = {}
                    for col in attr_cols:
                        if col in row.index:
                            val = row[col]
                            attrs[col] = "" if pd.isna(val) else str(val)
                    with xf.element(row_tag, **attrs):
                        # Sub-element per non-attribute column (or per-mapping column if mapping is non-empty)
                        if mapping:
                            for m in mapping:
                                col = m["column"]
                                if col in attr_cols:
                                    continue
                                if col in row.index:
                                    val = row[col]
                                    text = "" if pd.isna(val) else str(val)
                                    with xf.element(col):
                                        xf.write(text)
                        else:
                            for col in row.index:
                                if col in attr_cols:
                                    continue
                                val = row[col]
                                text = "" if pd.isna(val) else str(val)
                                with xf.element(col):
                                    xf.write(text)
                if flushonrow:
                    xf.flush()
                written += 1
            self._streaming_total_written += written

            # 8. globalMap puts (every chunk; final value is correct)
            if self.global_map:
                self.global_map.put(f"{self.id}_FILE_NAME", str(filepath))
                self.global_map.put(f"{self.id}_NB_LINE", self._streaming_total_written)

            # 9. Stats
            self._update_stats(written, written, 0)

            logger.info(
                "[%s] Wrote %d rows to %s (total=%d)",
                self.id, written, filepath, self._streaming_total_written,
            )

            # NOTE: We do NOT close the xmlfile context here -- the streaming hook
            # holds it across chunks. Closure happens in execute()'s finalize step
            # (which calls reset()) or when the engine tears down the component.
            # If this is the final / only chunk, the BaseComponent finalize hook
            # will trigger reset() and the contexts will close cleanly.

            return {"main": input_data, "reject": None}
    ```
