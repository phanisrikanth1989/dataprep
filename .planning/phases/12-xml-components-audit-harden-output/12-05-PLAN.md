    pattern: instead of `path.split("/")`, use a regex that respects bracket nesting:
    ```python
    import re
    _SEGMENT_RE = re.compile(r'(?:[^/\[]+(?:\[[^\]]*\])?)+')
    def split_steps(path: str) -> list:
        # Walk the string, balancing bracket depth, splitting only on '/' at depth 0.
        segments = []
        buf = []
        depth = 0
        for ch in path:
            if ch == '[':
                depth += 1
                buf.append(ch)
            elif ch == ']':
                depth = max(depth - 1, 0)
                buf.append(ch)
            elif ch == '/' and depth == 0:
                if buf:
                    segments.append("".join(buf))
                buf = []
            else:
                buf.append(ch)
        if buf:
            segments.append("".join(buf))
        return segments
    ```
