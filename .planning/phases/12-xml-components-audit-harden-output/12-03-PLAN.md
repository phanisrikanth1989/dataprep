    **Module structure (~350-500 LOC):**

    ```python
    """tFileInputXML engine component.

    Reads an XML file, applies a LOOP_QUERY XPath to identify repeating elements,
    extracts MAPPING-defined columns per element, and returns a DataFrame.

    Threshold-switched I/O (D-C2): files below xml_streaming_threshold_mb load via
    DOM (etree.parse), files at/above stream via etree.iterparse with element
    clearing. Secure parser flags (XXE / billion-laughs / no-network) per D-C4.
    Per-element MAPPING evaluation with REJECT routing for failures (ENG-FIX-002).
    """
    import logging
    import os
    from typing import Any, Dict, List, Optional, Tuple

    import pandas as pd
    from lxml import etree

    from . import _xml_io
    from ..base_component import BaseComponent
    from ..component_registry import REGISTRY
    from ..exceptions import ConfigurationError, FileOperationError

    logger = logging.getLogger(__name__)


    @REGISTRY.register("FileInputXML", "tFileInputXML")
    class FileInputXML(BaseComponent):
        """Read XML file -> DataFrame; threshold-switched DOM/streaming."""

        # Error code constants (S-3 reject schema)
        _ERR_FILE_MISSING = "FILE_MISSING"
        _ERR_PARSE = "PARSE_ERROR"
        _ERR_XPATH = "XPATH_ERROR"
        _ERR_NODECHECK = "NODECHECK_FAIL"

        def _validate_config(self) -> None:
            """Rule 12: presence + type checks only; content checks deferred to _process."""
            if not self.config.get("filename"):
                raise ConfigurationError(
                    f"[{self.id}] Missing required config key 'filename'"
                )
            if not self.config.get("loop_query"):
                raise ConfigurationError(
                    f"[{self.id}] Missing required config key 'loop_query'"
                )
            for bool_key, default in (
                ("die_on_error", False),
                ("ignore_ns", False),
                ("ignore_dtd", False),
                ("advanced_separator", False),
                ("check_date", False),
                ("use_separator", False),
            ):
                val = self.config.get(bool_key, default)
                if not isinstance(val, bool):
                    raise ConfigurationError(
                        f"[{self.id}] Config '{bool_key}' must be a boolean"
                    )

        def _process(self, input_data: Optional[pd.DataFrame] = None) -> Dict[str, Any]:
            filepath = self.config["filename"]
            loop_query = self.config["loop_query"]
            mapping = self.config.get("mapping", [])
            limit_raw = self.config.get("limit", "")
            die_on_error = self.config.get("die_on_error", False)
            ignore_ns = self.config.get("ignore_ns", False)
            encoding = self.config.get("encoding", "ISO-8859-15")
            threshold_mb = int(self.config.get("xml_streaming_threshold_mb", 50))

            # Resolve LIMIT semantics (ENG-FIX-007)
            #   ""     -> None (unlimited)
            #   "0"    -> 0 (read nothing)
            #   "N"    -> int(N)
            limit: Optional[int]
            if limit_raw == "" or limit_raw is None:
                limit = None
            else:
                try:
                    limit = int(limit_raw)
                except (TypeError, ValueError):
                    raise ConfigurationError(
                        f"[{self.id}] LIMIT must be empty or a non-negative integer; got {limit_raw!r}"
                    )

            # File existence check (deferred from _validate_config per Rule 12)
            if not os.path.exists(filepath):
                if die_on_error:
                    raise FileOperationError(
                        f"[{self.id}] XML file not found: {filepath!r}"
                    )
                return self._build_reject_only(self._ERR_FILE_MISSING, f"file not found: {filepath!r}")

            # Strategy decision + log (Pitfall P-4 mitigation)
            size_mb = os.stat(filepath).st_size / (1024 * 1024)
            try:
                strategy, parsed = _xml_io.parse_xml_strategy(filepath, threshold_mb)
            except etree.XMLSyntaxError as exc:
                if die_on_error:
                    raise FileOperationError(
                        f"[{self.id}] XML parse failed: {exc}"
                    ) from exc
                return self._build_reject_only(self._ERR_PARSE, str(exc))
            _xml_io.log_strategy(self.id, strategy, size_mb, threshold_mb)

            main_rows: List[Dict[str, Any]] = []
            reject_rows: List[Dict[str, Any]] = []

            if strategy == "dom":
                tree = parsed
                root = tree.getroot()
                # Pitfall P-5 mitigation: build per-tree nsmap from full tree iteration,
                # not just root.nsmap. Skip if ignore_ns.
                nsmap = self._build_nsmap(root, ignore_ns)
                # Apply LOOP_QUERY xpath
                loop_xpath = self._normalize_loop_query(loop_query, ignore_ns)
                try:
                    nodes = root.xpath(loop_xpath, namespaces=nsmap)
                except etree.XPathEvalError as exc:
                    if die_on_error:
                        raise FileOperationError(
                            f"[{self.id}] LOOP_QUERY XPath invalid: {loop_query!r}: {exc}"
                        ) from exc
                    return self._build_reject_only(self._ERR_XPATH, f"loop_query: {exc}")

                for idx, node in enumerate(nodes):
                    if limit is not None and idx >= limit:
                        break
                    self._extract_node(node, mapping, nsmap, main_rows, reject_rows, die_on_error)
            else:
                # streaming path
                # Extract loop_tag from loop_query: take last path segment, strip leading slashes
                loop_tag = loop_query.rstrip("/").split("/")[-1]
                loop_tag = loop_tag.removeprefix("@")  # P-7 use removeprefix not lstrip
                idx = 0
                for node in _xml_io.iterparse_loop_query(filepath, loop_tag):
                    if limit is not None and idx >= limit:
                        break
                    nsmap = self._build_nsmap(node, ignore_ns)
                    self._extract_node(node, mapping, nsmap, main_rows, reject_rows, die_on_error)
                    idx += 1

            col_names = [m["column"] for m in mapping] if mapping else []
            main_df = (
                pd.DataFrame(main_rows, columns=col_names) if main_rows else pd.DataFrame(columns=col_names)
            )
            reject_df = pd.DataFrame(reject_rows) if reject_rows else pd.DataFrame()
            total_rows = len(main_df) + len(reject_df)
            self._update_stats(total_rows, len(main_df), len(reject_df))
            logger.info(
                "[%s] done: file=%r ok=%d reject=%d strategy=%s",
                self.id, filepath, len(main_df), len(reject_df), strategy,
            )
            return {"main": main_df, "reject": reject_df}

        # ------------------------------------------------------------------
        # Helpers
        # ------------------------------------------------------------------

        def _build_nsmap(self, element: Any, ignore_ns: bool) -> Dict[str, str]:
            """Build a prefix->uri map from this element AND its ancestors+descendants.

            ENG-FIX-004 / Pitfall P-5: prior code read only root.nsmap. lxml exposes
            per-element nsmap but multi-namespace docs need a tree-wide collection.
            """
            if ignore_ns:
                return {}
            collected: Dict[str, str] = {}
            # walk down: include nsmaps from descendants
            try:
                for el in element.iter():
                    for k, v in (el.nsmap or {}).items():
                        if k and k not in collected:
                            collected[k] = v
            except Exception:
                pass
            return collected

        def _normalize_loop_query(self, query: str, ignore_ns: bool) -> str:
            """Apply ignore_ns transform: strip namespace prefixes from XPath if requested."""
            if not ignore_ns:
                return query
            # Strip "ns:" prefixes from path segments. Simple heuristic: replace
            # "/ns:tag" with "/*[local-name()='tag']" — Talend's IGNORE_NS=true effect.
            import re
            return re.sub(r"/(\w+):(\w+)", r"/*[local-name()='\2']", query)

        def _extract_node(
            self,
            node: Any,
            mapping: List[Dict[str, Any]],
            nsmap: Dict[str, str],
            main_rows: List[Dict[str, Any]],
            reject_rows: List[Dict[str, Any]],
            die_on_error: bool,
        ) -> None:
            """Evaluate MAPPING per node; route to main or reject. ENG-FIX-005 fix."""
            out_row: Dict[str, Any] = {}
            for m in mapping:
                col = m["column"]
                xpath_expr = m.get("xpath", "")
                nodecheck = m.get("nodecheck", False)
                try:
                    value = self._eval_mapping_xpath(node, xpath_expr, nsmap)
                except etree.XPathEvalError as exc:
                    if die_on_error:
                        raise FileOperationError(
                            f"[{self.id}] MAPPING XPath {xpath_expr!r} invalid: {exc}"
                        ) from exc
                    reject_rows.append({
                        "errorCode": self._ERR_XPATH,
                        "errorMessage": f"{xpath_expr}: {exc}",
                    })
                    return
                if nodecheck and (value is None or value == ""):
                    reject_rows.append({
                        "errorCode": self._ERR_NODECHECK,
                        "errorMessage": f"nodecheck failed for column {col!r} xpath {xpath_expr!r}",
                    })
                    return
                # ENG-FIX-005: explicitly None for missing columns; never silent zip-drop
                out_row[col] = value if value != "" else None
            main_rows.append(out_row)

        def _eval_mapping_xpath(self, node: Any, expr: str, nsmap: Dict[str, str]) -> Any:
            """Evaluate one MAPPING xpath; supports text() default and bare @attr."""
            if not expr:
                # default: take element text
                return node.text or ""
            # ENG-FIX-008: bare @attr means "attribute of the loop element"
            if expr.startswith("@"):
                return node.get(expr[1:], "")
            try:
                results = node.xpath(expr, namespaces=nsmap)
            except etree.XPathEvalError:
                raise
            if not results:
                return ""
            first = results[0]
            if hasattr(first, "text"):
                return first.text or ""
            return str(first)

        def _build_reject_only(self, code: str, msg: str) -> Dict[str, Any]:
            """Return {'main': empty, 'reject': single-row} for whole-file failures."""
            reject_df = pd.DataFrame([{"errorCode": code, "errorMessage": msg}])
            self._update_stats(1, 0, 1)
            return {"main": pd.DataFrame(), "reject": reject_df}
    ```
