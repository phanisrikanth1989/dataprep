    ```python
    """Shared helpers for XML engine components.

    Provides a hardened XMLParser factory (replaces deprecated defusedxml.lxml per
    Phase 12 RESEARCH.md P-1) and threshold-switched DOM/iterparse helpers used by
    every in-scope XML engine component.

    Public API:
        secure_xml_parser(recover=False) -> etree.XMLParser
        parse_xml_strategy(filename, threshold_mb) -> tuple[str, object]
        iterparse_loop_query(filename, loop_tag) -> Iterator[etree._Element]
        log_strategy(component_id, strategy, size_mb, threshold_mb) -> None

    Security: resolve_entities=False (XXE), no_network=True, load_dtd=False
    (billion-laughs vector). recover=False fails loud on malformed XML so callers
    can route to REJECT instead of silently passing partial trees.
    """
    import logging
    import os
    from typing import Iterator, Tuple

    from lxml import etree

    logger = logging.getLogger(__name__)


    def secure_xml_parser(*, recover: bool = False) -> etree.XMLParser:
        """Build a hardened XMLParser.

        Disables external entity expansion (XXE), DTD loading (billion-laughs vector),
        and network access. recover=False fails loud on malformed XML.
        """
        return etree.XMLParser(
            resolve_entities=False,
            no_network=True,
            load_dtd=False,
            recover=recover,
        )


    def parse_xml_strategy(filename: str, threshold_mb: int) -> Tuple[str, object]:
        """Decide DOM vs streaming based on file size.

        Returns:
            ('dom', etree._ElementTree) when size < threshold_mb (full parse)
            ('stream', filename) when size >= threshold_mb (caller calls
                iterparse_loop_query)

        Raises:
            FileNotFoundError / OSError if filename does not exist (propagated
                from os.stat — caller decides REJECT vs raise).
        """
        size_mb = os.stat(filename).st_size / (1024 * 1024)
        if size_mb < threshold_mb:
            tree = etree.parse(filename, parser=secure_xml_parser())
            return ("dom", tree)
        return ("stream", filename)


    def iterparse_loop_query(filename: str, loop_tag: str) -> Iterator[etree._Element]:
        """Yield each `loop_tag` end-event element; clear after consume.

        Memory-correct: clears the matched element AND walks back removing prior
        siblings so the prefix tree is released. Pitfall P-3 mitigation.
        """
        # iterparse cannot accept an XMLParser directly; the secure flags must be
        # passed as keyword args (lxml 4.x+).
        ctx = etree.iterparse(
            filename,
            events=("end",),
            tag=loop_tag,
            resolve_entities=False,
            no_network=True,
            load_dtd=False,
            recover=False,
        )
        for _event, element in ctx:
            yield element
            # 1. clear the matched element (free its subtree)
            element.clear(keep_tail=True)
            # 2. drop preceding siblings to release the prefix
            while element.getprevious() is not None:
                del element.getparent()[0]
        del ctx


    def log_strategy(
        component_id: str,
        strategy: str,
        size_mb: float,
        threshold_mb: int,
    ) -> None:
        """Emit a single ASCII-only INFO line so tests can spy on which branch ran.

        Pitfall P-4 mitigation: the threshold-switched code path must be observable.
        """
        logger.info(
            "[%s] XML strategy=%s size=%.2fMB threshold=%dMB",
            component_id, strategy, size_mb, threshold_mb,
        )
    ```

    ```python
    """Shared pytest fixtures for engine file-component tests.

    Provides synthetic_60mb_xml: a programmatically generated ~60 MB XML file
    used by streaming-path tests across plans 12-02 (this) and 12-03..12-07.
    The fixture is built fresh per session in tmp_path; nothing persists in
    the repo.
    """
    import os
    from pathlib import Path

    import pytest


    @pytest.fixture(scope="session")
    def synthetic_60mb_xml(tmp_path_factory) -> Path:
        """Build a ~60 MB XML file once per pytest session.

        Structure: <root><item id='N'><payload>{1KB}</payload></item> * ~60_000 </root>
        Used by streaming-path tests across plans 12-02..12-07.
        """
        path = tmp_path_factory.mktemp("xml_streaming") / "synthetic_60mb.xml"
        payload = "x" * 1000  # 1 KB per item
        with open(path, "w", encoding="utf-8") as f:
            f.write('<?xml version="1.0" encoding="UTF-8"?>\n<root>\n')
            for i in range(60_000):  # ~60_000 KB == ~60 MB
                f.write(f"<item><id>{i}</id><payload>{payload}</payload></item>\n")
            f.write("</root>\n")
        size_mb = os.stat(path).st_size / (1024 * 1024)
        assert 55 <= size_mb <= 65, f"synthetic XML size {size_mb}MB outside 55-65 band"
        return path
    ```

    ```python
    from lxml import etree

        <?xml version="1.0"?>
        <!DOCTYPE foo [
          <!ENTITY xxe SYSTEM "file:///etc/passwd">
        ]>
        <root>&xxe;</root>
    """).encode("utf-8")

    _BILLION_LAUGHS_DOC = textwrap.dedent("""\
        <?xml version="1.0"?>
        <!DOCTYPE lolz [
          <!ENTITY lol "lol">
          <!ENTITY lol1 "&lol;&lol;&lol;&lol;&lol;&lol;&lol;&lol;&lol;&lol;">
          <!ENTITY lol2 "&lol1;&lol1;&lol1;&lol1;&lol1;&lol1;&lol1;&lol1;&lol1;&lol1;">
          <!ENTITY lol3 "&lol2;&lol2;&lol2;&lol2;&lol2;&lol2;&lol2;&lol2;&lol2;&lol2;">
        ]>
        <lolz>&lol3;</lolz>
    """).encode("utf-8")


    @pytest.mark.unit
    class TestSecureXmlParser:
        def test_returns_xmlparser_instance(self):
            p = _xml_io.secure_xml_parser()
            assert isinstance(p, etree.XMLParser)

        def test_recover_default_false(self):
            # Malformed XML should raise, not silently pass partial tree
            p = _xml_io.secure_xml_parser()
            with pytest.raises(etree.XMLSyntaxError):
                etree.fromstring(b"<root><unclosed></root>", parser=p)

        def test_recover_true_swallows_malformed(self):
            p = _xml_io.secure_xml_parser(recover=True)
            tree = etree.fromstring(b"<root><unclosed></root>", parser=p)
            assert tree is not None

        def test_xxe_entities_not_resolved(self):
            p = _xml_io.secure_xml_parser()
            tree = etree.fromstring(_XXE_DOC, parser=p)
            # resolve_entities=False -- the &xxe; reference is NOT replaced with /etc/passwd
            text = (tree.text or "")
            assert "/etc/passwd" not in text
            assert "root:" not in text

        def test_billion_laughs_does_not_explode(self):
            p = _xml_io.secure_xml_parser()
            # libxml2's built-in expansion limit + load_dtd=False kills the bomb
            with pytest.raises(etree.XMLSyntaxError):
                etree.fromstring(_BILLION_LAUGHS_DOC, parser=p)

        def test_doctype_present_but_entities_unresolved(self):
            # A DOCTYPE declaration is allowed; entity references in body remain literal
            doc = b"<?xml version='1.0'?><!DOCTYPE r [<!ENTITY x 'y'>]><r>&x;</r>"
            p = _xml_io.secure_xml_parser()
            tree = etree.fromstring(doc, parser=p)
            # &x; is NOT expanded because resolve_entities=False
            assert "y" not in (tree.text or "")
    ```
