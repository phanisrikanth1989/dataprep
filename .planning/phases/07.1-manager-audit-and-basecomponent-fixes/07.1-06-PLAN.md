       if func_name == "median":
           if use_financial_precision:

               # Fall back to float median but warn so users aware of precision loss.
               if not getattr(self, "_warned_median_decimal", False):

                       f"not directly supported; falling back to float median (precision "

                   )
                   self._warned_median_decimal = True
               return lambda x: x.dropna().astype(float).median() if ignore_null \
