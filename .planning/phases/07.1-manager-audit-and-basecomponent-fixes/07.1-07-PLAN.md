    ```python
    """Engine component for Normalize (tNormalize).

    Splits a single column on a separator, exploding into multiple rows. Other
    columns are preserved unchanged.

    Config keys consumed (5 total):
      normalize_column           (str, required)             -- column to split
      item_separator             (str, default ",")          -- separator string (literal, not regex)
      trim                       (bool, default False)        -- strip whitespace from each split value
      discard_trailing_empty_str (bool, default False)        -- only TRAILING empties stripped (Talend semantics)
      deduplicate                (bool, default False)        -- dedupe split values per row
    """
    from __future__ import annotations
    import logging
    from typing import Any, Optional

    import pandas as pd

    from ...base_component import BaseComponent
    from ...component_registry import REGISTRY
    from ...exceptions import ConfigurationError

    logger = logging.getLogger(__name__)


    @REGISTRY.register("Normalize", "tNormalize")
    class Normalize(BaseComponent):
        """tNormalize engine implementation.

        Splits values in `normalize_column` by `item_separator`, exploding into
        multiple output rows. Other columns are preserved per Talend semantics.

        Talend reference: tNormalize_main.javajet
        (https://raw.githubusercontent.com/Talaxie/tdi-studio-se/master/main/plugins/
        org.talend.designer.components.localprovider/components/tNormalize/
        tNormalize_main.javajet)
        """

        def _validate_config(self) -> None:
            if "normalize_column" not in self.config:
                raise ConfigurationError(
                    f"[{self.id}] Missing required config key 'normalize_column'"
                )
            nc = self.config["normalize_column"]
            if not isinstance(nc, str) or not nc.strip():
                raise ConfigurationError(
                    f"[{self.id}] 'normalize_column' must be a non-empty string"
                )
            sep = self.config.get("item_separator", ",")
            if not isinstance(sep, str):
                raise ConfigurationError(
                    f"[{self.id}] 'item_separator' must be a string"
                )
            for b in ("deduplicate", "trim", "discard_trailing_empty_str", "die_on_error"):
                if b in self.config and not isinstance(self.config[b], bool):
                    raise ConfigurationError(
                        f"[{self.id}] '{b}' must be a boolean"
                    )

        def _process(self, input_data: Optional[pd.DataFrame] = None) -> dict:
            if input_data is None:
                return {"main": pd.DataFrame(), "reject": None}
            if input_data.empty:
                return {"main": input_data, "reject": None}

            norm_col = self.config["normalize_column"]
            sep = self.config.get("item_separator", ",")
            trim = self.config.get("trim", False)
            discard_trailing = self.config.get("discard_trailing_empty_str", False)
            dedupe = self.config.get("deduplicate", False)

            if norm_col not in input_data.columns:
                raise ConfigurationError(
                    f"[{self.id}] Column '{norm_col}' not found in input"
                )

            df = input_data.copy()
            # Vectorized split: maxsplit=-1 keeps trailing empties; we trim them later.
            splits = (
                df[norm_col]
                .astype("string")
                .fillna("")
                .str.split(sep, regex=False, n=-1)
            )

            # Talend ordering (per help.qlik.com tNormalize docs and Talaxie source):
            #   discard_trailing_empty_str -> trim -> dedupe
            if discard_trailing:
                splits = splits.apply(self._strip_trailing_empties)
            if trim:
                splits = splits.apply(lambda lst: [v.strip() for v in lst])
            if dedupe:
                splits = splits.apply(lambda lst: list(dict.fromkeys(lst)))

            df[norm_col] = splits

            # Empty-list rows: explode produces NaN for empty list, then we drop those
            # if discard_trailing is on (Talend emits 0 rows for fully consumed splits).
            out = df.explode(norm_col, ignore_index=True)
            if discard_trailing:
                # Drop rows where the normalize column is NaN (came from empty-list explode)
                out = out.dropna(subset=[norm_col]).reset_index(drop=True)
            else:
                out[norm_col] = out[norm_col].fillna("")

            return {"main": out, "reject": None}

        @staticmethod
        def _strip_trailing_empties(lst: list) -> list:
            """Mirror Talaxie tNormalize's lastNoEmptyIndex_ logic.

            Removes ONLY trailing empty strings, preserving interior empties.
            Talaxie source: lastNoEmptyIndex_ iterates from end, stops at first non-empty.

            Examples:
                _strip_trailing_empties(["a", "b", "", ""]) == ["a", "b"]
                _strip_trailing_empties(["a", "", "b", ""]) == ["a", "", "b"]
                _strip_trailing_empties(["", "", ""]) == []
            """
            if not lst:
                return lst
            last_idx = len(lst)
            while last_idx > 0 and lst[last_idx - 1] == "":
                last_idx -= 1
            return lst[:last_idx]
    ```
