    ```python
    def _route_catch_output_rejects(
        self,
        result: dict,
        raw_result: dict,
        outputs_config: list[dict],
    ) -> None:
        """Route expression error rows to catch output reject outputs.

        User-defined column expressions on the catch output are evaluated
        per row against the joined_df row context.  Two column names are
        framework-reserved (D-06):
          - errorMessage: populated with the exception message text.
          - errorStackTrace: populated with the stack trace string.
        If the user declares these names with their own expressions, the
        framework value WINS (matches Talaxie tMap_main.inc.javajet:1925-1988).

        Args:
            result: Current result dict to update.
            raw_result: Raw result from compiled script execution.
            outputs_config: Output config list.
        """
        error_key = "__errors__"
        if error_key not in raw_result:
            return

        error_df = raw_result[error_key]
        if error_df is None or (isinstance(error_df, pd.DataFrame) and error_df.empty):
            return

        # Extract per-row error messages from raw_result if available.
        # The bridge encodes error messages in "__error_messages__" (a list
        # parallel to error_df rows) when catch_output_reject is configured.
        error_messages = raw_result.get("__error_messages__", [])
        error_traces = raw_result.get("__error_traces__", [])

        main_name = self.config["inputs"]["main"]["name"]
        lookup_names_list = [lk["name"] for lk in self.config["inputs"].get("lookups", [])]

        for output_cfg in outputs_config:
            if not output_cfg.get("catch_output_reject"):
                continue
            out_name = output_cfg["name"]

            # Evaluate user-defined column expressions against error rows (D-01).
            # Skip errorMessage / errorStackTrace during evaluation (D-06).
            # Build a filtered config that excludes the reserved column names.
            non_reserved_cols = [
                c for c in output_cfg["columns"]
                if c["name"] not in ("errorMessage", "errorStackTrace")
            ]
            eval_cfg = {**output_cfg, "columns": non_reserved_cols}
            user_df = self._evaluate_output_columns_py(
                error_df if isinstance(error_df, pd.DataFrame) else pd.DataFrame([error_df]),
                eval_cfg,
                {},
                main_name,
                lookup_names_list,
            )

            # Re-add reserved columns with framework-supplied values (D-06).
            col_names = [c["name"] for c in output_cfg["columns"]]
            if "errorMessage" in col_names:
                if error_messages:
                    user_df["errorMessage"] = (
                        error_messages[:len(user_df)]
                        + ["Expression evaluation error"] * max(0, len(user_df) - len(error_messages))
                    )
                else:
                    user_df["errorMessage"] = "Expression evaluation error"
            if "errorStackTrace" in col_names:
                if error_traces:
                    user_df["errorStackTrace"] = (
                        error_traces[:len(user_df)]
                        + [""] * max(0, len(user_df) - len(error_traces))
                    )
                else:
                    user_df["errorStackTrace"] = ""

            result[out_name] = user_df
            logger.info(
                "[%s] Routed %d error rows to catch output '%s'",
                self.id, len(user_df), out_name,
            )
    ```

    Note: The raw_result "__error_messages__" / "__error_traces__" availability depends on the bridge output format. If the bridge does not currently produce these keys, use a default value. Document any discrepancy in SUMMARY.
