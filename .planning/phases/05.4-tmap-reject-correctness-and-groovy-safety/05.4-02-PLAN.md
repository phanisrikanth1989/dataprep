       ```python
       class _NullRow:
           """Sentinel for a lookup table row that was not matched in an inner join.

           Unlike _NullNamespace (which returns None for any attribute to mirror
           left-outer semantics), _NullRow raises AttributeError to signal that
           a column reference on a failed inner-join lookup is an expression error.
           The error propagates through the existing die_on_error / catch_output_reject
           machinery -- matching Talend inner-join reject semantics (D-04).
           """

           def __getattr__(self, name: str) -> None:
               raise AttributeError(
                   f"lookup was not matched; column '{name}' unavailable on reject row"
               )

           def __setattr__(self, name: str, value: Any) -> None:
               pass  # discard -- null rows are read-only
       ```

    2. Rewrite `_route_inner_join_rejects` (map.py:2583-2626):
       ```python
       def _route_inner_join_rejects(
           self,
           result: dict,
           inner_join_reject_dfs: dict[str, pd.DataFrame],
           outputs_config: list[dict],
       ) -> None:
           if not inner_join_reject_dfs:
               return

           all_rejects = pd.concat(
               list(inner_join_reject_dfs.values()), ignore_index=True
           )
           if all_rejects.empty:
               return

           for output_cfg in outputs_config:
               if not output_cfg.get("inner_join_reject"):
                   continue
               out_name = output_cfg["name"]
               # Evaluate reject output's own column expressions against the
               # unmatched main rows (D-01, D-02). var_columns={} -- variable
               # values are not computed at reject-routing time.
               reject_df = self._evaluate_output_columns_py(
                   all_rejects,
                   output_cfg,
                   {},  # var_columns
                   self.config["inputs"]["main"]["name"],
                   [lk["name"] for lk in self.config["inputs"].get("lookups", [])],
               )
               if out_name in result and not result[out_name].empty:
                   result[out_name] = pd.concat(
                       [result[out_name], reject_df], ignore_index=True
                   )
               else:
                   result[out_name] = reject_df
               logger.info(
                   "[%s] Routed %d inner join rejects to output '%s'",
                   self.id, len(reject_df), out_name,
               )
       ```
       Note: `self.config["inputs"]["main"]["name"]` and the lookup names are read from self.config since _route_inner_join_rejects is called from _process which has already validated these.
