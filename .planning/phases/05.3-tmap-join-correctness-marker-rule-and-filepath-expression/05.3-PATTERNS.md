**New method shape (planner reference):**
```python
# Locality buckets (replace _JOIN_EQUALITY / _JOIN_CONTEXT_ONLY / _JOIN_CROSS_TABLE constants)
_LOCALITY_CONTEXT = "context"        # only context/globalMap/Var
_LOCALITY_MAIN_SIDE = "main_side"    # only main + previously-joined lookups (anything in joined_df)
_LOCALITY_LOOKUP_SIDE = "lookup_side"  # only the current lookup
_LOCALITY_TWO_SIDED = "two_sided"    # both sides

def _classify_key_locality(
    self, expr: str, main_name: str, current_lookup: str,
    joined_lookup_names: list[str],
) -> str:
    stripped = self._strip_java_marker(expr).strip()
    matches = _ROW_REF_PATTERN.findall(stripped)
    row_refs = [m[0] for m in matches if m[0] not in ("context", "globalMap", "Var")]
    if not row_refs:
        return _LOCALITY_CONTEXT
    main_side_tables = {main_name, *joined_lookup_names}  # joined_df contains all of these
    refs_main = any(t in main_side_tables and t != current_lookup for t in row_refs)
    refs_lookup = current_lookup in row_refs
    if refs_main and refs_lookup:
        return _LOCALITY_TWO_SIDED
    if refs_lookup:
        return _LOCALITY_LOOKUP_SIDE
    return _LOCALITY_MAIN_SIDE
```

**New method shape (planner reference, combining both):**
```python
def _chunked_cross_product(
    self,
    main_df: pd.DataFrame,
    lookup_df: pd.DataFrame,
    match_expr: Optional[str],   # filter or two-sided join expression; None = pure cartesian
    main_name: str,
    lookup_name: str,
    joined_lookup_names: list[str],
) -> pd.DataFrame:
    if lookup_df.empty:
        return main_df.iloc[0:0]
    chunk_size = self._compute_cross_chunk_size(len(lookup_df))
    result_chunks: list[pd.DataFrame] = []
    for start in range(0, len(main_df), chunk_size):
        chunk = main_df.iloc[start:start + chunk_size]
        chunk_cross = pd.merge(chunk, lookup_df, how="cross")
        if match_expr is None:
            result_chunks.append(chunk_cross)
            continue
        eval_results = self._evaluate_with_bridge(
            chunk_cross, {"__match__": match_expr},
            main_name, joined_lookup_names + [lookup_name],   # D-04: full lookup list
        )
        if "__match__" in eval_results:
            mask = pd.Series(eval_results["__match__"]).fillna(False).values
            result_chunks.append(chunk_cross[mask])
    if not result_chunks:
        return main_df.iloc[0:0]
    return pd.concat(result_chunks, ignore_index=True)

@staticmethod
def _compute_cross_chunk_size(lookup_rows: int) -> int:
    return max(100, min(10_000, 100_000_000 // max(1, lookup_rows)))
```
