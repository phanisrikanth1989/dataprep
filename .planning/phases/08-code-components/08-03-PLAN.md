       ```python
       def _process(self, input_data: Optional[pd.DataFrame] = None) -> dict[str, Any]:
           if input_data is None or input_data.empty:
               return {"main": input_data, "reject": None}

           java_code = self.config.get("java_code", "")
           imports = self.config.get("imports", "")
           output_schema = self.config.get("output_schema")

           if imports:
               java_code = imports + "\n" + java_code  # D-07/D-08

           if not self.java_bridge:
               raise ComponentExecutionError(
                   self.id,
                   "Java execution requested but no Java bridge available",
               )

           logger.debug(
               f"[{self.id}] Executing per-row Java code on {len(input_data)} rows "
               f"(code size={len(java_code)} chars)"
           )
           try:
               main_df = self.java_bridge.execute_java_row(
                   df=input_data,
                   java_code=java_code,
                   output_schema=output_schema,
               )
               return {"main": main_df, "reject": None}
           except Exception as e:
               # Q2 Option A: batch-level reject. The bridge protocol surfaces a single
               # exception for the whole batch (JavaBridge.java:243-246). We route the
               # entire input batch to reject with the bridge error message.
               if self.die_on_error:
                   raise ComponentExecutionError(
                       self.id,
                       f"Java per-row execution failed (die_on_error=True): {e}",
                       cause=e,
                   ) from e

               error_message = f"{e.__class__.__name__}: {e}"
               logger.warning(
                   f"[{self.id}] Java per-row execution failed; routing entire "
                   f"batch of {len(input_data)} rows to reject (Q2 Option A): "
                   f"{error_message}"
               )
               reject_df = input_data.copy()
               reject_df["errorMessage"] = error_message  # D-16
               reject_df["errorCode"] = 1                 # D-16 -- int, not string
               main_df = pd.DataFrame(columns=input_data.columns)
               return {"main": main_df, "reject": reject_df}
       ```
